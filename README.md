## Install

```sh
$ npm install ctrip-apollo --save
```

## Usage

```js
import ap from 'ctrip-apollo'

ap({
  server,   //http://${ip}:${port}
  appId,
  namespace = 'application',
  cluster = 'default',
  ip
})
.then((apollo) => {
  const a = apollo.get('a')
  const {c, d} = apollo.get('c', 'd')
})
```
